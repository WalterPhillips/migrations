16    private static final String MIGRATIONS_PROPERTIES = "migration.properties";

18    public NewCommand(SelectedOptions options) {
19      super(options);

22    public void execute(String... params) {
23      if (paramsEmpty(params)) {
24        throw new MigrationException("No description specified for new migration.");
25      }
26      String description = params[0];
27      Properties variables = new Properties();
28      variables.setProperty("description", description);
29      existingEnvironmentFile();
30      String filename = getNextIDAsString() + "_" + description.replace(' ', '_') + ".sql";
31      String migrationsHome = "";
32      migrationsHome = System.getenv(MIGRATIONS_HOME);
33  
34      // Check if there is a system property
35      if (migrationsHome == null) {
36        migrationsHome = System.getProperty(MIGRATIONS_HOME_PROPERTY);
37      }
38  
39      if (options.getTemplate() != null) {
40        copyExternalResourceTo(options.getTemplate(), Util.file(paths.getScriptPath(), filename));
41      } else if ((migrationsHome != null) && (!migrationsHome.equals(""))) {
42        try {
43          //get template name from properties file
44          final String customConfiguredTemplate =
45              ExternalResources.getConfiguredTemplate(migrationsHome + "/" + MIGRATIONS_PROPERTIES,
46                  CUSTOM_NEW_COMMAND_TEMPATE_PROPERTY);
47          copyExternalResourceTo(migrationsHome + "/" + customConfiguredTemplate,
48              Util.file(paths.getScriptPath(), filename));
49        } catch (FileNotFoundException e) {
50          printStream.append(
51              "Your migrations configuration did not find your custom template.  Using the default template.");
52          copyDefaultTemplate(variables, filename);
53        }
54      } else {
55        copyDefaultTemplate(variables, filename);
56      }
57  
58      printStream.println("Done!");
59      printStream.println();
60    }
61  
62    private void copyDefaultTemplate(Properties variables, String filename) {
63      copyResourceTo("org/apache/ibatis/migration/template_migration.sql",
64          Util.file(paths.getScriptPath(), filename),
65          variables);
66    }
