21    public FileMigrationLoader(File scriptsDir, String charset, Properties properties) {
22      super();

28    @Override
29    public List<Change> getMigrations() {
30      List<Change> migrations = new ArrayList<Change>();
31      if (scriptsDir.isDirectory()) {
32        String[] filenames = scriptsDir.list();
33        if (filenames == null) {
34          throw new MigrationException(scriptsDir + " does not exist.");
35        }
36        Arrays.sort(filenames);
37        for (String filename : filenames) {
38          if (filename.endsWith(".sql") && !"bootstrap.sql".equals(filename)) {
39            Change change = parseChangeFromFilename(filename);
40            migrations.add(change);
41          }
42        }
43      }
44      return migrations;
45    }
46  
47    private Change parseChangeFromFilename(String filename) {
48      try {
49        Change change = new Change();
50        String[] parts = filename.split("\\.")[0].split("_");
51        change.setId(new BigDecimal(parts[0]));
52        StringBuilder builder = new StringBuilder();
53        for (int i = 1; i < parts.length; i++) {
54          if (i > 1) {
55            builder.append(" ");
56          }
57          builder.append(parts[i]);
58        }
59        change.setDescription(builder.toString());
60        change.setFilename(filename);
61        return change;
62      } catch (Exception e) {
63        throw new MigrationException("Error parsing change from file.  Cause: " + e, e);
64      }
65    }
66  
67    @Override
68    public Reader getScriptReader(Change change, boolean undo) {
69      try {
70        return new MigrationReader(Util.file(scriptsDir, change.getFilename()), charset, undo, properties);
71      } catch (IOException e) {
72        throw new MigrationException("Error reading " + change.getFilename(), e);
73      }
74    }
75  
76    @Override
77    public Reader getBootstrapReader() {
78      try {
79        File bootstrap = Util.file(scriptsDir, "bootstrap.sql");
80        if (bootstrap.exists()) {
81          return new MigrationReader(bootstrap, charset, false, properties);
82        }
83        return null;
84      } catch (IOException e) {
85        throw new MigrationException("Error reading bootstrap.sql", e);
86      }
